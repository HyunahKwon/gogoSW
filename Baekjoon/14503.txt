int MAX = 0;

void selectSort(int arr[]) {

	for (int i = 0; i < arr_size - 1; i++)
	{
		MAX = i;
		for (int j = i + 1; j < arr_size; j++) {
			if (arr[j] >= arr[MAX]) {
				MAX = j;

			}
		}
		int max_count = 0;
		for (int j = i; j < arr_size - 1; j++) {
			if (arr[j] == arr[j + 1] && arr[j] == arr[MAX]) max_count = j; //ÃÖ¼Ò 1
		}

		int des = 0;
		if (arr[i] > arr[i + 1])  des++;
		
		if (des > 0 && max_count != 0)  swap(arr[i], arr[MAX - 1]);
		else swap(arr[i], arr[MAX]);


		if (i != MAX)	current_change_num++;

		if (current_change_num == input_change_num) {
			check = 1;
			break;
		}
	}



}
